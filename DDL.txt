
DATA TYPES:
NUMERIC:
INT---INTEGERS---ID---DEPTNO---AGE---PINCODES---SALARY
DECIMAL(A,B)--- FOR DECIMAL POINT---SALARY(6,2)---5000.00

STRING:
VARCHAR(SIZE)---- VARCHAR(20)-----NAMES---SHARANYA
CHAR(SIZE)----CHAR(8)---FIXED SIZE---SHARANYA

DATE AND TIME:
DATE---FOR DATE---HIREDATE---LOGTIME---'YYYY-MM-DD'
TIME----FOR TIME ---HIRETIME---'HH-MM-SS'


DDL COMMANDS

CREATE,CHANGES(MODIFY),RENAME,DROP



COMMANDS
1.CREATE: FOR CREATE TABLE,DATABASE
FOR DATABASE:
SYNTAX:

CREATE DATABASE DATABASE_NAME;

EXAMPLE:
CREATE DATABASE FULL_STACK;

FOR TABLE:
SYNTAX:
CREATE TABLE TABLE_NAME(CLOUMN(S) DATATYPE(S));

EX:
CREATE TABLE STUDENTS(ID INT,NAME VARCHAR(30),AGE INT,EMAIL VARCHAR(50));

 
2.ALTER: FOR MODIFY,RENAME,DROP,ADD 

SYNTAX:
ALTER TABLE TABLE_NAME CMD COLUMN CLOUMN_NAME DATATYPE;



ADD CLOUMN: ADD COLUMN TO THE TABLE
SYNTAX:
ALTER TABLE TABLE_NAME ADD COLUMN COLUMN_NAME DATATYPE;

EX:
ALTER TABLE STUDENTS ADD COLUMN COURSE VARCHAR(30);


MODIFY: TO MODIFY TABLE

SYNTAX:
ALTER TABLE TABLE_NAME MODIFY CLOUMN COLUMN_NAME DATATYPE;


EX:
ALTER TABLE STUDENTS MODIFY CLOUMN COURSE VARCHAR(50);

RENAME: TO RENAME THE COLUMN NAME

SYNTAX:
ALTER TABLE TABLE_NAME RENAME CLOUMN OLD_COLUMN_NAME TO NEW_COLUMN_NAME;


EX:
ALTER TABLE STUDENTS RENAME CLOUMN NAME TO SNAME ;


DROP: TO DROP COLUMN

SYNTAX:
ALTER TABLE TABLE_NAME DROP CLOUMN COLUMN_NAME ;


EX:
ALTER TABLE STUDENTS DROP CLOUMN EMAIL;



3.RENAME:RENAME THE TABLE NAME
SYNTAX:
RENAME TABLE OLD_TABLE_NAME TO NEW_TABLE_NAME;
EX:
RENAME TABLE STUDENTS TO STD;


4.TRUNCATE:TO DELETE ALL ROWS IN THE TABLE

SYNTAX:
TRUNCATE TABLE TABLE_NAME;

EXAMPLE:
TRUNCATE TABLE STUDENTS;


5.DROP: TO DELETE TABLE FROM THE DATABASE COMPLETELY

SYNTAX:
DROP TABLE TABLE_NAME;

EXAMPLE:
DROP TABLE STUDENTS;
